% package for including graphics with figure-environment
\usepackage{graphicx}
% For subfigure
\usepackage{subcaption}

% Add the margin around the caption
\usepackage[margin=1cm]{caption}

\newcommand{\image}[4][1]{
\begin{figure}[ht!]
  \includegraphics[trim=0 0 0 -2cm,width=#1\textwidth,keepaspectratio]{#2}
  \centering
  \caption{\normalsize#3}
  \label{#4}
\end{figure}
} 
% Example usage
% \image[0.7]{img/Skein_permutation.png}{Hoán vị Skein \
% \citep{wiki:skein_permutation}}{fig:skein}

\newcommand{\doubleimage}[7][0.4]{ 
\begin{figure}[ht!]
  \centering
  \begin{subfigure}{.5\textwidth}
    \centering
    \includegraphics[width=#1\linewidth]{#2}
    \caption{\normalsize#3}
  \end{subfigure}%
  \begin{subfigure}{.5\textwidth}
    \centering
    \includegraphics[width=#1\linewidth]{#4}
    \caption{\normalsize#5}
  \end{subfigure}
  \caption{#6}
  \label{#7}
\end{figure} 
}
